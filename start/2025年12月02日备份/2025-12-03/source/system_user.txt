[[module UserInfo]]

------

此页面用于快捷封禁用户。

此工具需要安装油猴脚本（见最底部）

首先在下方文本域输入用户名，可以是{{@@[[*user 用户名]]@@}}、{{@@[[user 用户名]]@@}}或{{用户名}}的格式。可以以换行来批量输入

然后点击确定，再按提示分别封禁。

    <script>
        function processInput() {
            const input = document.getElementById('inputText').value;
            const lines = input.split('\n');
            
            const processedLines = lines.map(line => {
                let processed = line.replace(/^\[\[\*?user\s+/, '').trim();
                processed = processed.replace(/\]\]$/, '').trim();
                processed = processed.replace(/[ _]/g, '-');
                return processed;
            }).filter(line => line.length > 0);
            
            processedLines.forEach(username => {
                const url = `https://if-backrooms.wikidot.com/system:user/${username}`;
                window.open(url, '_blank');
            });
        }
    </script>
